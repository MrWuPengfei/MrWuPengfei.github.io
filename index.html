                    %s
"""; //3.创建计时器任务 Timer timer = new Timer(); TimerTask task = new TimerTask(){ @Override public void run() { // 更新内容并显示 int i = 0; while (i < texts.size()) { String currentText = String.format(htmlTemplate,texts.get(i)); Jsoup.parseBodyElements(currentText).get(0).prepend(); i++; try { Thread.sleep(500);// 每500毫秒更新一次 } catch (InterruptedException e){ e.printStackTrace(); } } } }; //4.启动定时器任务 timer.schedule(task，0，50); } } '''
